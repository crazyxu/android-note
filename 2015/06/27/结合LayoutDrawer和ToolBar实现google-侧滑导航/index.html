MainActivity.java

主要代码，如有疑问或建议可以在网页底部反馈
package me.xucan.android_note;
import android.os.Bundle;
import android.support.v4.app.Fragment;
import android.support.v4.app.FragmentTransaction;
import android.support.v4.widget.DrawerLayout;
import android.support.v7.app.ActionBarDrawerToggle;
import android.support.v7.app.AppCompatActivity;
import android.support.v7.widget.Toolbar;
import android.view.Menu;
import android.view.MenuItem;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ArrayAdapter;
import android.widget.LinearLayout;
import android.widget.ListView;

import me.xucan.android_note.image_util.ImageUtilFragment;
import me.xucan.android_note.simulate_login.SimulateLoginFragment;


public class MainActivity extends AppCompatActivity{
    //ToolBar
    private Toolbar toolbar;
    //drawer
    private DrawerLayout drawer;
    //却换箭头
    private ActionBarDrawerToggle drawerToggle;

    //侧滑layout
    private LinearLayout leftDrawer;

    //功能列表
    private ListView lv;

    //列表数据
    private String[] arr_module;

    //静态常量
    private final static int FRA_MAIN=0;
    private final static int FRA_IMAGE_UTIL=1;
    private final static int FRA_SIMULATE_LOGIN=2;


    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        initToolBar();
        initDrawer();
        initLeftLayout();
    }

    //初始化ToolBar
    void initToolBar(){
        toolbar=(Toolbar)super.findViewById(R.id.tool_bar);
        setSupportActionBar(toolbar);
    }

    //初始化DrawerLayout
    void initDrawer(){
        drawer=(DrawerLayout)super.findViewById(R.id.drawer);
        //drawer监听事件
        drawerToggle=new ActionBarDrawerToggle(this,drawer,toolbar,R.string.open_drawer,R.string.close_drawer){
            @Override
            public void onDrawerClosed(View drawerView) {
                super.onDrawerClosed(drawerView);
            }

            @Override
            public void onDrawerOpened(View drawerView) {
                super.onDrawerOpened(drawerView);
            }
        };
        //Sync the toggle state after onRestoreInstanceState has occurred.
        drawerToggle.syncState();
        drawer.setDrawerListener(drawerToggle);
        //绑定默认fragment,MainFragment
        android.support.v4.app.FragmentTransaction ft=getSupportFragmentManager().beginTransaction();
        ft.replace(R.id.fragment_layout, new MainFragment());
        ft.commit();
    }

    //初始化左侧导航
    void initLeftLayout(){
        leftDrawer=(LinearLayout)super.findViewById(R.id.drawer_left_layout);
        initDrawerLv();
    }

    //初始化侧滑菜单列表
    void initDrawerLv(){
        lv=(ListView)super.findViewById(R.id.drawer_lv);
        //列表数据
        arr_module=getResources().getStringArray(R.array.arr_module);
        //适配器
        ArrayAdapter adapter=new ArrayAdapter(this,android.R.layout.simple_list_item_1,arr_module);
        lv.setAdapter(adapter);
        //事件监听
        lv.setOnItemClickListener(new ListView.OnItemClickListener() {
            //点击item,切换到对应Fragment
            @Override
            public void onItemClick(AdapterView<?> adapterView, View view, int i, long l) {
                Fragment fra = null;
                switch (i) {
                    case FRA_MAIN:
                        fra = new MainFragment();
                        break;
                    case FRA_IMAGE_UTIL:
                        fra = new ImageUtilFragment();
                        break;
                    case FRA_SIMULATE_LOGIN:
                        fra = new SimulateLoginFragment();
                        break;
                }
                FragmentTransaction ft = getSupportFragmentManager().beginTransaction();
                ft.replace(R.id.fragment_layout, fra);
                ft.commit();
                //更改toolbar标题
                toolbar.setTitle(arr_module[i]);
                //关闭drawer
                drawer.closeDrawer(leftDrawer);
            }
        });
    }



    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        // Inflate the menu; this adds items to the action bar if it is present.
        getMenuInflater().inflate(R.menu.menu_main, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(MenuItem item) {
        switch (item.getItemId()){
            case R.id.action_settings:
                break;
            case android.R.id.home:
                //通过箭头（Home）来打开或关闭drawer
                if(drawer.isDrawerOpen(leftDrawer))
                    drawer.closeDrawer(leftDrawer);
                else
                    drawer.openDrawer(leftDrawer);
                break;
        }
        return super.onOptionsItemSelected(item);
    }

}

其他代码

在arrays.xml和strings.xml中自行添加,以上是显示类似google+的效果，实现知乎的效果只需要修改activity_main.xml，将ToolBar放在DrawerLayout之外，如下

<LinearLayout
    xmlns:android="http://schemas.android.com/apk/res/android"
    android:layout_width="fill_parent"
    android:layout_height="fill_parent"
    android:orientation="vertical">
    <android.support.v7.widget.Toolbar
        android:layout_width="fill_parent"
        android:layout_height="wrap_content"
        android:background="#ff12b8dd"
        android:id="@+id/tool_bar">
    </android.support.v7.widget.Toolbar>
    <android.support.v4.widget.DrawerLayout
        android:layout_width="fill_parent"
        android:layout_height="fill_parent"
        android:id="@+id/drawer">
        <FrameLayout
            android:layout_width="fill_parent"
            android:layout_height="fill_parent"
            android:id="@+id/fragment_layout">
        </FrameLayout>
        <include
            android:id="@+id/drawer_left_layout"
            layout="@layout/drawer_left_layout"
            />
    </android.support.v4.widget.DrawerLayout>
</LinearLayout>

注意

  1. 本文使用的开发工具是Android Studio，如果你在eclipse下开发，必须手动添加android-support-v7-appcompat库项目，切记不能简单的添加v4和v7的jar文件，因为我们需要用到v7库项目下的一些其他资源。
  2. 代码中用到的Fragment，FragmentTransaction，DrawerLayout，Toolbar等，必须使用v4,v7兼容包下的类。
  3. 使用Android Studio开发环境创建的Fragment文件，实现了onAttach方法，该代码告诉我们需要在使用该Fragment的Activity中实现OnFragmentInteractionListener接口，否则会报错，此接口用于Fragment与Activity的交互，如果不需要可将如下代码删除。
@Override
public void onAttach(Activity activity) {
    super.onAttach(activity);
    try {
        mListener = (OnFragmentInteractionListener) activity;
    } catch (ClassCastException e) {
        throw new ClassCastException(activity.toString()
                + " must implement OnFragmentInteractionListener");
    }
}
